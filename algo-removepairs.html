프로그래머스 - 짝지어 제거하기

def solution(s):
    answer=1
    stack = []
    
    for i in s:
        if stack and i==stack[-1]:
            stack.pop()
            continue
            
        stack.append(i)
            
    if stack:
        answer=0
        
    return answer
